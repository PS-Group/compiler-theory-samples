def sqrt(x)
  if x < 0
    return 0
  end
  root = 1
  do
    newRoot = 0.5 * (root + x / root)
    mustContinue = !(newRoot == root)
    root = newRoot
  while mustContinue end
  return root
end

print "sqrt(2) = ", sqrt(2)
print "sqrt(3) = ", sqrt(3)
print "sqrt(9) = ", sqrt(9)
print "sqrt(10) = ", sqrt(10)
